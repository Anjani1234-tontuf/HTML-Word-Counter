<!DOCTYPE html>
<html>
<head>
	<script>
		function main(){
		var text = document.getElementById("inputBox").value;
		var characters = text.length;
		var words = 1;
		var i;
		var previousLetter;
			for (i = 0; i < characters; i++){
				var currentLetter = text.substr(i , 1);
					if (currentLetter == " "){
					if (previousLetter != currentLetter){
					words++
					}
					}
					previousLetter = currentLetter;
					}
					document.getElementById("characters").innerHTML = "Total Characters: " +characters;
					document.getElementById("words").innerHTML = "Total Words: " +words;
					}
					
	</script>
	</head>
	<body>
	<textarea id = "inputBox" onkeyup = "main();" placeholder = "Enter Some Text..."></textarea>
			<div id = "characters"></div>
			<div id = "words"></div>
	</body>
</html
